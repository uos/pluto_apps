cmake_minimum_required(VERSION 2.8.3)
project(dem_costmap_2d)

set(PACKAGE_DEPENDENCIES 
  roscpp
  pcl_ros
  pcl_conversions
  nav_msgs
  std_msgs
  std_srvs
  octomap_ros
  octomap_msgs
  dynamic_reconfigure
  pluginlib
  costmap_2d
)

find_package(catkin REQUIRED COMPONENTS ${PACKAGE_DEPENDENCIES})
find_package(PCL REQUIRED)
find_package(octomap REQUIRED)

find_package(Eigen REQUIRED)
find_package(Boost REQUIRED COMPONENTS system thread)

add_definitions(-DOCTOMAP_NODEBUGOUT)

include_directories(
    include
    ${catkin_INCLUDE_DIRS}
    ${EIGEN_INCLUDE_DIRS}
    ${PCL_INCLUDE_DIRS}
    ${Boost_INCLUDE_DIRS}
    ${OCTOMAP_INCLUDE_DIRS}
)

add_definitions(${EIGEN_DEFINITIONS})

# dynamic reconfigure
generate_dynamic_reconfigure_options(
  cfg/DEMCostmap2D.cfg
)

catkin_package(
    INCLUDE_DIRS
        include
        ${EIGEN_INCLUDE_DIRS}
        ${PCL_INCLUDE_DIRS}
        ${OCTOMAP_INCLUDE_DIRS}
    LIBRARIES costmap_2d
    CATKIN_DEPENDS
        dynamic_reconfigure
        pluginlib
        costmap_2d
        roscpp
        std_msgs
        octomap_msgs
    DEPENDS
        PCL
        Eigen
        Boost
        octomap
)


add_library(dem_costmap_2d src/dem_costmap_2d.cpp
)

add_dependencies(dem_costmap_2d ${PROJEKT_NAME}_gencfg)

link_libraries(dem_costmap_2d
  ${PCL_LIBRARIES}
  ${Boost_LIBRARIES}
  ${OCTOMAP_LIBRARIES}
  ${catkin_LIBRARIES}
)

link_directories(${catkin_LIBRARY_DIRS})

install(TARGETS dem_costmap_2d
  ARCHIVE DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

# Install headers
install(DIRECTORY include/${PROJECT_NAME}/
  DESTINATION ${CATKIN_PACKAGE_INCLUDE_DESTINATION}
)

install(FILES dem_costmap_2d_plugin.xml
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
